## 四、行业增强模块代码

```python
def calc_industry_score_v4(code, date, trend_w=20, crowd_w=60, mom_w=60):
    """
    V4: 综合评分（趋势+拥挤+动量+波动）
    
    返回值范围: [-1, 1]
    - >= 0.5: 偏多
    - <= -0.3: 偏空
    - 其他: 中性
    """
    try:
        prices = get_price(code, end_date=str(date), count=max(crowd_w, mom_w)+1,
                          fields=["close", "volume"], panel=False)
        if len(prices) < crowd_w:
            return 0
        
        close = prices["close"]
        vol = prices["volume"]
        
        # 趋势分
        trend_score = 1 if close.iloc[-1] > close.rolling(trend_w).mean().iloc[-1] else -1
        
        # 拥挤分
        vol_ratio = vol.iloc[-1] / vol.rolling(crowd_w).mean().iloc[-1]
        crowd_score = -1 if vol_ratio > 2.0 else 1
        
        # 动量分
        mom_score = 1 if (close.iloc[-1] / close.iloc[-mom_w] - 1) > 0 else -1
        
        # 波动率过滤
        ret = close.pct_change().dropna()
        vol_ratio_20 = ret.iloc[-20:].std() / ret.iloc[-crowd_w:].std()
        vol_score = -1 if vol_ratio_20 > 1.5 else 1
        
        return 0.3 * trend_score + 0.25 * crowd_score + 0.25 * mom_score + 0.2 * vol_score
    except:
        return 0


def industry_enhanced_filter(codes, date, threshold=0.3):
    """
    行业增强过滤器（可插拔模块）
    
    参数:
        codes: ETF代码列表
        date: 当前日期
        threshold: 过滤阈值，默认0.3
    
    返回:
        过滤后的ETF代码列表
    """
    scores = {c: calc_industry_score_v4(c, date) for c in codes}
    eligible = [c for c, sc in scores.items() if sc >= threshold]
    
    # 如果过滤后不足3只，降级为全池
    if len(eligible) < 3:
        return codes
    
    return eligible
```
