| 结构名称 | 开盘涨跌幅范围 | 定义说明 | 预期特征 |
|----------|----------------|----------|----------|
| 假弱高开 | +0.5% ~ +1.5% | 看似高开但涨幅有限，可能被误判为弱 | 弱转强逻辑，反包潜力大 |

def check_buy_signal_main_version(stock, date, prev_date):
    """
    主版本买点检查 - 假弱高开
    条件：昨日涨停 + 今日假弱高开 + 硬过滤 + 情绪过滤（可选）
    """
    # 1. 昨日涨停检查
    prev_data = get_price(stock, end_date=prev_date, count=1, 
                          fields=['close', 'high_limit'], panel=False)
    if prev_data.empty:
        return False
    prev_close = float(prev_data['close'].iloc[0])
    prev_high_limit = float(prev_data['high_limit'].iloc[0])
    if abs(prev_close - prev_high_limit) / prev_high_limit > 0.01:
        return False
    
    # 2. 今日假弱高开（+0.5%~+1.5%）
    curr_data = get_price(stock, end_date=date, count=1, 
                          fields=['open', 'close'], panel=False)
    if curr_data.empty:
        return False
    curr_open = float(curr_data['open'].iloc[0])
    open_pct = (curr_open - prev_close) / prev_close * 100
    if not (0.5 <= open_pct <= 1.5):
        return False
    
    # 3. 硬过滤：流通市值50-150亿
    q = query(valuation.code, valuation.circulating_market_cap).filter(
        valuation.code == stock)
    df = get_fundamentals(q, date=date)
    if df.empty:
        return False
    market_cap = float(df['circulating_market_cap'].iloc[0])
    if not (50 <= market_cap <= 150):
        return False
    
    # 4. 硬过滤：相对位置≤30%（15日内）
    prices = get_price(stock, end_date=date, count=15, 
                       fields=['close'], panel=False)
    if prices.empty:
        return False
    high_15d = float(prices['close'].max())
    low_15d = float(prices['close'].min())
    position = (prices['close'].iloc[-1] - low_15d) / (high_15d - low_15d)
    if position > 0.30:
        return False
    
    # 5. 硬过滤：近1日无涨停（避免连板接力）
    lb_data = get_price(stock, end_date=prev_date, count=2, 
                        fields=['close', 'high_limit'], panel=False)
    if len(lb_data) >= 2:
        prev_prev_close = float(lb_data['close'].iloc[-2])
        prev_prev_limit = float(lb_data['high_limit'].iloc[-2])
        if abs(prev_prev_close - prev_prev_limit) / prev_prev_limit < 0.01:
            return False
    
    # 6. 情绪过滤（可选）：涨停家数≥50
    # 情绪过滤可显著提升收益，建议启用
    # zt_count = get_zt_count(prev_date)
    # if zt_count < 50:
    #     return False
    
    return True
