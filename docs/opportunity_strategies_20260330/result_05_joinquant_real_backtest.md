### 6.2 实施代码

**完整的风控代码示例**：

```python
class RiskManager:
    """
    双层风控管理器
    第一层：连亏停手
    第二层：广度+情绪风控
    """
    def __init__(self):
        # 连亏停手参数
        self.loss_count = 0
        self.pause_days = 0
        
        # 广度情绪参数
        self.breadth_threshold = 0.15
        self.sentiment_threshold = 5
    
    def check_pause_mechanism(self, pnl_pct):
        """第一层：连亏停手检查"""
        if pnl_pct < 0:
            self.loss_count += 1
        else:
            self.loss_count = 0
        
        if self.loss_count >= 3 and self.pause_days == 0:
            self.pause_days = 3
            return "连亏停手"
        
        return None
    
    def check_market_environment(self, breadth, sentiment):
        """第二层：市场环境检查"""
        if breadth < self.breadth_threshold:
            return "极端行情清仓"
        elif sentiment < self.sentiment_threshold:
            return "低情绪减仓"
        
        return None
    
    def can_trade(self, breadth, sentiment):
        """综合判断是否可以交易"""
        # 检查连亏停手
        if self.pause_days > 0:
            self.pause_days -= 1
            return False, "停手中"
        
        # 检查市场环境
        market_signal = self.check_market_environment(breadth, sentiment)
        if market_signal == "极端行情清仓":
            return False, market_signal
        
        return True, "正常"
```
