from sklearn.svm import SVC
from sklearn.model_selection import cross_val_score
import pandas as pd
import numpy as np

class SVMStockSelector:
    def __init__(self, C=1.0, kernel='rbf'):
        self.model = SVC(probability=True, kernel=kernel, C=C, random_state=42)
        self.feature_cols = [
            'EP', 'BP', 'SP', 'CFP', 'pe_ratio', 'pb_ratio',
            'roe', 'roa', 'gross_profit_margin', 'net_profit_to_total_revenue',
            'inc_net_profit_year_on_year', 'inc_revenue_year_on_year',
            'log_market_cap'
        ]
    
    def train(self, X, y):
        """训练模型"""
        self.model.fit(X[self.feature_cols], y)
        return self
    
    def predict_proba(self, X):
        """预测上涨概率"""
        return self.model.predict_proba(X[self.feature_cols])[:, 1]
    
    def select_stocks(self, X, top_pct=0.3):
        """选股：返回概率最高的top_pct股票"""
        proba = self.predict_proba(X)
        pred_df = pd.DataFrame({'code': X.index, 'prob': proba})
        pred_df = pred_df.sort_values('prob', ascending=False)
        top_n = max(int(len(pred_df) * top_pct), 1)
        return pred_df.head(top_n)['code'].tolist()
```

# square(delta(div(5, high)))
# 即: (5/最高价的变化量)^2

def alpha_factor_1(stock, date):
    """遗传算法挖掘的最佳因子"""
    prices = get_price(stock, end_date=date, count=20, fields=['high'])
    high = prices['high']
    ratio = 5 / high
    delta = ratio.diff()
    factor = delta ** 2

from gplearn.genetic import SymbolicTransformer
from gplearn.functions import make_function

class GeneticFactorMiner:
    def __init__(self, population_size=3000, generations=5):
        self.est_gp = SymbolicTransformer(
            population_size=population_size,
            generations=generations,
            metric='spearman',
            tournament_size=30,
            random_state=42,
            verbose=1,
            parsimony_coefficient=0.0001,
            n_jobs=-1
        )
    
    def fit(self, X, y):
        """挖掘因子"""
        self.est_gp.fit(X, y)
        return self
    
    def get_best_factors(self, n=10):
        """获取最优因子"""
        return self.est_gp._best_programs[:n]
    
    def transform(self, X):
        """生成因子值"""
        return self.est_gp.transform(X)

from sklearn.cluster import KMeans
from sklearn.manifold import TSNE
import numpy as np

class StockClusterer:
    def __init__(self, n_clusters=10):
        self.n_clusters = n_clusters
        self.kmeans = KMeans(n_clusters=n_clusters, random_state=42)
    
    def fit(self, returns_matrix):
        """基于收益率聚类"""
        self.kmeans.fit(returns_matrix.T)
        return self
    
    def get_cluster_stocks(self, cluster_id, stock_list):
        """获取某类的股票"""
        labels = self.kmeans.labels_
        return [s for s, l in zip(stock_list, labels) if l == cluster_id]
    
    def get_cluster_center_stocks(self, returns_matrix, stock_list):
        """获取每类中心股票"""
        from sklearn.metrics import pairwise_distances_argmin_min
        centers = self.kmeans.cluster_centers_
        center_indices, _ = pairwise_distances_argmin_min(centers, returns_matrix.T)
        return [stock_list[i] for i in center_indices]
```

## 二、特征工程标准流程

### 标准化特征预处理

```python
def process_features(df):
    """
    特征预处理标准流程
    1. 去极值 (1%/99%分位数)
    2. 标准化 (Z-score)
    """
    # 1. 去极值
    for col in df.select_dtypes(include=[np.number]).columns:
        q1 = df[col].quantile(0.01)
        q99 = df[col].quantile(0.99)
        df[col] = df[col].clip(q1, q99)
    
    # 2. 标准化
    df = (df - df.mean()) / df.std()
    return df.fillna(0)
```

def monthly_rebalance(context, selector):
    """
    月度调仓函数
    在每月最后一个交易日执行
    """
    date = context.current_dt.strftime('%Y-%m-%d')
    
    # 1. 获取股票池
    stocks = get_index_stocks('000905.XSHG', date)  # 中证500

    # 3. 模型预测
    selected = selector.select_stocks(features, top_pct=0.3)
    
    # 4. 等权配置
    target_weight = 1.0 / len(selected)
    
    # 5. 调仓
    for stock in context.portfolio.positions:
        if stock not in selected:
            order_target(stock, 0)
    
    for stock in selected:
        order_target_value(stock, context.portfolio.total_value * target_weight)

def stop_loss_check(context, stop_loss_pct=0.1):
    """止损检查"""
    for stock, position in context.portfolio.positions.items():
        if position.avg_cost > 0:
            pnl = (position.price - position.avg_cost) / position.avg_cost
            if pnl < -stop_loss_pct:
                order_target(stock, 0)  # 触发止损
