# V5: 小市值选股策略（简化版）
# 经典量化策略，基本面+市值过滤
# 用于验收测试：验证 get_fundamentals, filter_st, filter_paused 等API

from jqdata import *
import pandas as pd

def initialize(context):
    # 设置参数
    g.index = '000300.XSHG'  # 沪深300作为股票池
    g.num_stocks = 10  # 持仓股票数量
    g.stocks = []

    log.info('小市值选股策略初始化')
    log.info(f'基准指数: {g.index}')
    log.info(f'持仓数量: {g.num_stocks}')

    # 设置定时器：每月首日开盘调仓
    run_monthly(rebalance, 1, 'open')

def rebalance(context):
    """月度调仓：选出小市值、低估值的股票"""

    log.info(f'月度调仓: {context.current_dt}')

    # 1. 获取指数成分股作为候选池
    stocks = get_index_stocks(g.index)
    log.info(f'候选股票池: {len(stocks)}只')

    # 2. 基本面过滤：选择小市值、低估值股票
    df = get_fundamentals(
        query(
            valuation.code,
            valuation.market_cap,  # 市值（亿元）
            valuation.pe_ratio,    # 市盈率
            valuation.pb_ratio,    # 市净率
        ).filter(
            valuation.code.in_(stocks),
            valuation.market_cap > 0,      # 市值有效
            valuation.pe_ratio > 0,        # 市盈率有效
            valuation.pe_ratio < 30,       # 低估值：市盈率<30
            valuation.pb_ratio > 0,        # 市净率有效
            valuation.pb_ratio < 3,        # 低估值：市净率<3
        )
    )

    if df is None or len(df) == 0:
        log.warning('基本面查询结果为空，无法选股')
        return

    # 3. 按市值排序，选择最小的前N只
    df = df.sort_values('market_cap', ascending=True)
    g.stocks = list(df['code'].head(g.num_stocks))

    log.info(f'选出股票: {g.stocks}')
    log.info(f'\n{df.head(g.num_stocks)}')

    # 4. 过滤ST股票、停牌股票
    current_data = get_current_data()
    tradable_stocks = []

    for stock in g.stocks:
        if current_data[stock].paused:
            log.info(f'过滤停牌股票: {stock}')
            continue

        # 注意：filter_st需要在策略代码中手动检查（通过current_data.is_st）
        # 这里简化处理，实际应调用 filter_st(g.stocks)

        tradable_stocks.append(stock)

    if len(tradable_stocks) == 0:
        log.warning('没有可交易股票')
        return

    # 5. 调仓：卖出不在持仓的股票
    for stock in context.portfolio.positions:
        if stock not in tradable_stocks:
            log.info(f'卖出: {stock} {current_data[stock].name}')
            order_target(stock, 0)

    # 6. 买入新股票
    position_value = context.portfolio.total_value / len(tradable_stocks)

    for stock in tradable_stocks:
        if stock not in context.portfolio.positions:
            if context.portfolio.available_cash > position_value:
                log.info(f'买入: {stock} {current_data[stock].name}, 金额{position_value:.0f}')
                order_value(stock, position_value)

# end
